\usepackage{libertine}

\usepackage[autocompile]{gregoriotex}

% Use this command to switch between simple/solumn styles
\newcommand{\antiphonstyle}{simple}

\begin{document}

{\LARGE Marian Antiphons}

\vspace{20mm}

\P~~\textit{From Advent 1 to Candlemas}

\gregorioscore{alma-redemptoris-mater-\antiphonstyle}

\vspace{3mm}

\textit{During advent:}

\Vbar. Angelus Dómini nuntiávit Maríae.

\Rbar. Et concépit de Spíritu Sáncto.

\newpage

\P~~\textit{From Candlemas to Easter Vigil}

\gregorioscore{ave-regina-caelorum-\antiphonstyle}

\vspace{3mm}

\Vbar. Dignáre me laudáre te Virgo sacráta

\Rbar. Da míhi virtútem contra hóstes túos.

\newpage

\P~~\textit{From Easter Vigil to Pentecost}

\gregorioscore{regina-caeli-\antiphonstyle}

\vspace{3mm}

\Vbar. Gáude et laetáre Vírgo María, allelúia.

\Rbar. Quia surréxit Dóminus vere. allelúia.

\newpage

\P~~\textit{From Pentecost to Advent 1}

\gregorioscore{salve-regina-\antiphonstyle}

\vspace{3mm}

\Vbar. Ora pro nóbis sáncta Déi Génitrix.

\Rbar. Ut digni efficiámur promissiónibus Chrísti.
